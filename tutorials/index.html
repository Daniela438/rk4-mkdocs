\[ f(t, \mathbf{y}) = -{\rm{i}} [\mathbf{O}, \mathbf{y}(t)] \]

El paquete de dynamics es muy útil para resolver este problema.

def dyn_generator(oper, state):
    """Implementa función f(y,t). Toma el operador y estado inicial
    Examples:
        >>> dyn_generator(oOper, yInit)
        ([0-0j 0+0j],[0-1j 0-0j])

    Args:
        oper (matrix): array bidimensional que modifica el estado y(t)
        state (matrix): array bidimensional que representa el estado inicial alguna cantidad física.
    Returns:
        (matrix): matriz que representa el conmutador de dos operadores por el número cmplejo `1j`.
    """
    return -iConst * (np.dot(oper, state) - np.dot(state, oper))

def rk4(func, oper, state, h):
    """Implementa la función que realiza el método RK4.
    Exemples:
        >>> rk4(dyn_generator, oOper, yInit, h)

    Args:
        func (function): función que representa la dinámica del problema
        oper (matrix): operador que modifica el estado de y(t)
        state (matrix): estado inicial
        h (float): paso temporal
    Returns:
        (matrix): el nuevo estado del sistema despues de un paso temporal h.
    """
    k1 = h * func(oper, state)
    k2 = h * func(oper, state + 0.5 * k1)
    k3 = h * func(oper, state + 0.5 * k2)
    k4 = h * func(oper, state + k3)

    return state + (1/6) * (k1 + 2 * k2 + 2 * k3 + k4)
